#include<iostream>
using namespace std;
int n;
int a[10];
bool ed[10];

void dfs(int u) {
    if(u==n) {
        for(int i=0;i<n;i++) cout<<a[i]<<" ";
        cout<<endl;
        return;
    }
    for(int i=1;i<=n;i++) {
        if(!ed[i]) {
            ed[i]=1;
            a[u]=i;
            dfs(u+1);
            a[u]=0;
            ed[i]=0; // 逐渐递归进入更深层，直到到达所建树的最低层之后然后再返回
        }
    }
}

int main() {
    cin>>n;
    dfs(0); // 先从0开始进行搜索
}
