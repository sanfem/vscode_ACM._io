单调栈：单调栈是一种特殊的栈结构，栈中的元素保持**单调递增或单调递减**的顺序。
### 常见应用场景

1. 下一个更大/更小元素问题
    
2. 柱状图中最大矩形面积
    
3. 接雨水问题
    
4. 滑动窗口最大值/最小值
```cpp
#include<bits/stdc++.h>

using namespace std;

stack<pair<int,int>> qp;

vector<int> a,ans(3e6+1);

int main()

{

    ios::sync_with_stdio(false);

    cin.tie(0);

    cout.tie(0);

  

    int n;

    cin>>n;

    for(int i=1;i<=n;i++)

    {

        int x;

        cin>>x;

        while(!qp.empty()&&qp.top().first<x)

        {

            pair<int,int> tem=qp.top();

            qp.pop();

            ans[tem.second]=i;

        }

        qp.push({x,i});

    }

    ans[qp.top().second]=0;

    for(int i=1;i<=n;i++) cout<<ans[i]<<" ";

}
```

### 常见应用场景

1. 滑动窗口最大值/最小值
    
2. 限制长度的队列最值问题
    
3. 优化动态规划问题
```cpp
 #include<bits/stdc++.h>

 using namespace std;

 typedef long long ll;

 deque<pair<ll,ll>> qp,Mqp;

 vector<ll> ans,M_ans;

 int main(){

    ios::sync_with_stdio(false);

    cin.tie(0);

    cout.tie(0);

  

    int n,m;

    cin>>n>>m;

    for(int i=1;i<=n;i++)

    {

        int x;

        cin>>x;

        while(!qp.empty()&&i-qp.front().second>m){

            qp.pop_front();

        }

        if(i>=m+1) ans.push_back(qp.front().first);

        while(!qp.empty()&&x<=qp.back().first)

        {

            qp.pop_back();

        }

        qp.push_back({x,i});

        while(!Mqp.empty()&&i-Mqp.front().second>m){

            Mqp.pop_front();

        }

        if(i>=m+1) M_ans.push_back(Mqp.front().first);

        while(!Mqp.empty()&&x>=Mqp.back().first)

        {

            Mqp.pop_back();

        }

        Mqp.push_back({x,i});

    }

    while(!qp.empty()&&n-qp.front().second>=m){

        qp.pop_front();

    }

    ans.push_back(qp.front().first);

    while(!Mqp.empty()&&n-Mqp.front().second>=m){

        Mqp.pop_front();

    }

    M_ans.push_back(Mqp.front().first);

    for(int i=0;i<=n-m;i++) cout<<ans[i]<<" ";

    cout<<'\n';

    for(int i=0;i<=n-m;i++) cout<<M_ans[i]<<" ";

    cout<<'\n';

    return 0;

 }
```
