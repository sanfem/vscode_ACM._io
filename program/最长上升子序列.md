Lcs:用 二分法优化
```cpp
#include<bits/stdc++.h>

using namespace std;

vector<int> a,b;

void Lower_bounder(int x){

    int l=0,r=b.size()-1,mid;

    int ans;

    while(l<=r)

    {

        mid= l+r >>1;

        if(x<=b[mid]){

            ans=mid;

            r=mid-1;

        }

        else{

            l=mid+1;

        }  

    }

    b[ans]=x;

}

int main(){

    ios::sync_with_stdio(false);

    cin.tie(0);

    cout.tie(0);

  

    int n;

    cin>>n;

    for(int i=0;i<n;i++){

        int x;

        cin>>x;

        a.push_back(x);

    }

    b.push_back(a[0]);

    for(int i=1;i<n;i++){

        if(a[i]>b[b.size()-1]) b.push_back(a[i]);

        else Lower_bounder(a[i]);

    }

    cout<<b.size();

}
```
