```cpp
 const MAX_N=1<<20;
 int H[MAX_N+1];
 for(int i=0;i<=20;i++) H[1<<i]=i;
 while(cin>>n){
	 while(n>0){
	 cout<<H[n&n-1]<<'';
	 n -= n & -n;
	 }
	 cout<<end1;
 }
```
