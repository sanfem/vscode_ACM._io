**并查集**是一种树型的数据结构，用于处理一些不交集合的合并及查询问题。
初始化：让每一个数据指向自身。
查找：Find（）；
合并：UNION（）；
```cpp
#include<bits/stdc++.h>

using namespace std;

const int MAXN=1e6+1;

vector<int> fa(MAXN);

int Find(int x){

    if(x==fa[x]) return x;

    else{

        fa[x]=Find(fa[x]);

    }

    return fa[x];

}

void Union(int x,int y)

{   int Fx=Find(x),Fy=Find(y);

    if(Fx==Fy){

        return;

    }

    else{

        fa[Fx]=Fy;

    }

}

int main(){

    ios::sync_with_stdio(false);

    cin.tie(0);

    cout.tie(0);

  

    int n,m;

    cin>>n>>m;

    for(int i=1;i<=n;i++){

        fa[i]=i;

    }

    for(int i=1;i<=m;i++){

        int x,y,z;

        cin>>z>>x>>y;

        if(z==1){

            Union(x,y);

        }

        else{

            if(Find(x)==Find(y)) cout<<"Y"<<'\n';

            else cout<<"N"<<'\n';

        }

    }

}
```
已经对上面的问题采取路径压缩，提高代码效率。可用与解决连通性问题
