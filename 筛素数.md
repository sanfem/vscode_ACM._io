#include<algorithm>
using namespace std;
int prime[10000005] = {0};
void isprime(int n)
{
	for(int i = 2; i <=n; i++){
		if(prime[i] == 0){
			for(int j = 2; j * i <= n; j++){
				prime[i * j] = 1;
				cout << i * j <<" ";
			}cout << endl;
		}
	}
	for(int i = 2; i <= n; i++){
		if(prime[i] == 0){
			cout << i <<" ";
		}
	}
}
